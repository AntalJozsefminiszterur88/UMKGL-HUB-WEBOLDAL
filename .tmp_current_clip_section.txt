 13569:       function formatBytes(bytes) {
 13570:         if (!Number.isFinite(bytes) || bytes <= 0) return "0 B";
 13571:         const units = ["B", "KB", "MB", "GB", "TB"];
 13572:         let value = bytes;
 13573:         let unitIndex = 0;
 13574:         while (value >= 1024 && unitIndex < units.length - 1) {
 13575:           value /= 1024;
 13576:           unitIndex += 1;
 13577:         }
 13578:         const precision = value >= 10 || unitIndex === 0 ? 0 : 1;
 13579:         return `${value.toFixed(precision)} ${units[unitIndex]}`;
 13580:       }

 14018: 
 14019:         const trimmed = rawTitle.trim();
 14020:         const looksLikeMojibake = /[\u00C3\u00C2]/.test(trimmed);
 14021:         if (looksLikeMojibake) {
 14022:           const decoded = new TextDecoder("utf-8").decode(
 14023:             Uint8Array.from(trimmed.split("").map((char) => char.charCodeAt(0)))
 14024:           );
 14025:           if (decoded && !decoded.includes("\uFFFD")) {
 14026:             return decoded.replace(/\.[^.]+$/i, "").trim();
 14027:           }
 14028:         }

 14053:             return clipNameTagPriority.get(firstName) - clipNameTagPriority.get(secondName);
